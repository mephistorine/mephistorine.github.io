        class GCounterImpl implements GCounter {
          private value: number
          constructor(initialValue: number = 0) {
            this.value = initialValue
          }
          public getValue(): number {
            return this.value
          }
        }
        class GCounterImpl implements GCounter {
          // ...
          public increment(num: number): void {
            this.value = this.value + num
          }
          public merge(otherCounter: GCounter): void {
            const otherCounterValue = otherCounter.getValue()
            this.value = Math.max(this.value, otherCounterValue)
          }
        }
